func PopCount2(x uint64) int {
    var res int
    for i := 1; i < 8; i++ {
        res += int(pc[byte(x>>(i*8))])
    }
    return res
}

// count each bit
func PopCount3(x uint64) int {
    var res int
    for x > 0 {
        if x&1 == 1 {
            res++
        }
        x >>= 1
    }
    return res
}
// x & (x - 1)
func PopCount4(x uint64) int {
    var res int
    for x > 0 {
        x &= (x - 1)
        res++
    }
    return res
}
